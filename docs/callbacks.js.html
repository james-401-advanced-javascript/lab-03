'use strict';

const fs = require('fs');
const faker = require('faker');

/**
 *
 * @param {json} file
 * @function read
 */
function read(file) {
  // We want to read the file and then do some error-first handling of the response
  let path = `${process.cwd()}/${file}`;
  fs.readFile(path, function(err, data) {
    // If error, handle error
    if (err) {
      throw err;
    } else {
      // If no error, assign job title to person
      data = JSON.parse(data);
      console.log('Original Callback Data: ', data);
      data.jobTitle = faker.name.jobTitle();
      data = JSON.stringify(data);
    }

    // Write changes to data and verify changes
    fs.writeFile(path, data, err => {
      if (err) {
        throw err;
      } else {
        console.log('New Callback Data: ', JSON.parse(data));
      }
    });
  });
}

module.exports = read;
